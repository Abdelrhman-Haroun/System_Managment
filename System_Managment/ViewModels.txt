    public class TransactionListViewModel
    {
        public int Id { get; set; }
        public string TransactionNumber { get; set; }
        public string Type { get; set; }
        public string ProductName { get; set; }
        public decimal Quantity { get; set; }

        public string Party { get; set; }

        public DateTime CreatedDate { get; set; }
    }

    public class TransactionCreateViewModel
    {
        [Required]
        [Display(Name = "Type")]
        public string Type { get; set; }

        [Required]
        [Display(Name = "Product")]
        public int ProductId { get; set; }

        [Required]
        [Display(Name = "Quantity")]
        public decimal Quantity { get; set; }

        [Required]
        [Display(Name = "Unit Price")]
        public decimal UnitPrice { get; set; }

        [Display(Name = "Customer")]
        public int? CustomerId { get; set; }

        [Display(Name = "Supplier")]
        public int? SupplierId { get; set; }

        [Display(Name = "Cashbox")]
        public int? CashboxId { get; set; }

        [Display(Name = "Notes")]
        public string Notes { get; set; }

        // Dropdowns
        public List<ProductSelectViewModel> Products { get; set; } = new List<ProductSelectViewModel>();
        public List<CustomerSelectViewModel> Customers { get; set; } = new List<CustomerSelectViewModel>();
        public List<SupplierSelectViewModel> Suppliers { get; set; } = new List<SupplierSelectViewModel>();
        public List<CashboxSelectViewModel> Cashboxes { get; set; } = new List<CashboxSelectViewModel>();
    }
