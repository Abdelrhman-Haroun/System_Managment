using System.ComponentModel.DataAnnotations;

public class Product
{
    public int Id { get; set; }
    [Required]
    public string Name { get; set; }
    public decimal Price { get; set; }
}

public class Customer
{
    public int Id { get; set; }
    [Required]
    public string Name { get; set; }
    public decimal Balance { get; set; } = 0; // Debt balance
}

public class Supplier
{
    public int Id { get; set; }
    [Required]
    public string Name { get; set; }
    public decimal Balance { get; set; } = 0; // Debt balance
}

public class Employee
{
    public int Id { get; set; }
    [Required]
    public string Name { get; set; }
    public decimal Salary { get; set; }
}

public class Cashbox
{
    public int Id { get; set; }
    [Required]
    public string Name { get; set; } = "Main Cashbox";
    public decimal Balance { get; set; } = 0;
}

// Transaction Model (for Sales and Purchases)
public class Transaction
{
    public int Id { get; set; }
    [Required]
    public string Type { get; set; } // "Sale" or "Purchase"
    public int? CustomerId { get; set; } // For Sales
    public int? SupplierId { get; set; } // For Purchases
    public int ProductId { get; set; }
    public int Quantity { get; set; }
    public decimal Price { get; set; }
    public decimal Total { get; set; } // Calculated: Quantity * Price
    public decimal PaidAmount { get; set; } = 0;
    public decimal Remaining { get; set; } // Calculated: Total - PaidAmount
    public DateTime Date { get; set; } = DateTime.Now;
    public string Notes { get; set; }

    public Product Product { get; set; }
    public Customer Customer { get; set; }
    public Supplier Supplier { get; set; }
}

// CashboxTransaction Model (for Cash Movements)
public class CashboxTransaction
{
    public int Id { get; set; }
    [Required]
    public string Type { get; set; } // "Payment", "Expense", "Salary", "Other"
    public int? CustomerId { get; set; } // For payments from customers
    public int? SupplierId { get; set; } // For payments to suppliers
    public int? EmployeeId { get; set; } // For salaries
    public decimal Amount { get; set; } // Positive for income, negative for expense
    public DateTime Date { get; set; } = DateTime.Now;
    public string Notes { get; set; }
    public int CashboxId { get; set; } = 1;

    public Customer Customer { get; set; }
    public Supplier Supplier { get; set; }
    public Employee Employee { get; set; }
    public Cashbox Cashbox { get; set; }
}

// ApplicationDbContext
public class ApplicationDbContext : DbContext
{
    public ApplicationDbContext(DbContextOptions<ApplicationDbContext> options) : base(options) { }

    public DbSet<Product> Products { get; set; }
    public DbSet<Customer> Customers { get; set; }
    public DbSet<Supplier> Suppliers { get; set; }
    public DbSet<Employee> Employees { get; set; }
    public DbSet<Cashbox> Cashboxes { get; set; }
    public DbSet<Transaction> Transactions { get; set; }
    public DbSet<CashboxTransaction> CashboxTransactions { get; set; }
}



















using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;
using System.Threading.Tasks;
using System.Linq;

public class TransactionsController : Controller
{
    private readonly ApplicationDbContext _context;

    public TransactionsController(ApplicationDbContext context)
    {
        _context = context;
    }

    // GET: Create Sale/Purchase Transaction
    public IActionResult Create()
    {
        ViewBag.Customers = _context.Customers.ToList();
        ViewBag.Suppliers = _context.Suppliers.ToList();
        ViewBag.Products = _context.Products.ToList();
        return View();
    }

    // POST: Save Sale/Purchase Transaction
    [HttpPost]
    [ValidateAntiForgeryToken]
    public async Task<IActionResult> Create(Transaction transaction)
    {
        if (ModelState.IsValid)
        {
            transaction.Total = transaction.Quantity * transaction.Price;
            transaction.Remaining = transaction.Total - transaction.PaidAmount;

            // Update customer or supplier balance
            if (transaction.Type == "Sale" && transaction.CustomerId.HasValue)
            {
                var customer = await _context.Customers.FindAsync(transaction.CustomerId.Value);
                if (customer != null)
                {
                    customer.Balance += transaction.Remaining;
                    _context.Update(customer);

                    // Record cash movement if paid
                    if (transaction.PaidAmount > 0)
                    {
                        var cashbox = await _context.Cashboxes.FindAsync(1); // Default cashbox
                        if (cashbox != null)
                        {
                            cashbox.Balance += transaction.PaidAmount;
                            _context.Update(cashbox);

                            _context.CashboxTransactions.Add(new CashboxTransaction
                            {
                                Type = "Payment",
                                CustomerId = transaction.CustomerId,
                                Amount = transaction.PaidAmount,
                                Notes = $"Payment for Sale #{transaction.Id}",
                                CashboxId = 1
                            });
                        }
                    }
                }
            }
            else if (transaction.Type == "Purchase" && transaction.SupplierId.HasValue)
            {
                var supplier = await _context.Suppliers.FindAsync(transaction.SupplierId.Value);
                if (supplier != null)
                {
                    supplier.Balance += transaction.Remaining;
                    _context.Update(supplier);

                    // Record cash movement if paid
                    if (transaction.PaidAmount > 0)
                    {
                        var cashbox = await _context.Cashboxes.FindAsync(1);
                        if (cashbox != null)
                        {
                            cashbox.Balance -= transaction.PaidAmount;
                            _context.Update(cashbox);

                            _context.CashboxTransactions.Add(new CashboxTransaction
                            {
                                Type = "Payment",
                                SupplierId = transaction.SupplierId,
                                Amount = -transaction.PaidAmount,
                                Notes = $"Payment for Purchase #{transaction.Id}",
                                CashboxId = 1
                            });
                        }
                    }
                }
            }

            _context.Add(transaction);
            await _context.SaveChangesAsync();
            return RedirectToAction("Index");
        }

        ViewBag.Customers = _context.Customers.ToList();
        ViewBag.Suppliers = _context.Suppliers.ToList();
        ViewBag.Products = _context.Products.ToList();
        return View(transaction);
    }

    // GET: Create Cashbox Transaction (Payment/Expense/Salary)
    public IActionResult CreateCashTransaction()
    {
        ViewBag.Customers = _context.Customers.ToList();
        ViewBag.Suppliers = _context.Suppliers.ToList();
        ViewBag.Employees = _context.Employees.ToList();
        ViewBag.Cashboxes = _context.Cashboxes.ToList();
        return View();
    }

    // POST: Save Cashbox Transaction
    [HttpPost]
    [ValidateAntiForgeryToken]
    public async Task<IActionResult> CreateCashTransaction(CashboxTransaction cashTransaction)
    {
        if (ModelState.IsValid)
        {
            var cashbox = await _context.Cashboxes.FindAsync(cashTransaction.CashboxId);
            if (cashbox == null) return NotFound();

            if (cashTransaction.Amount == 0)
            {
                ModelState.AddModelError("Amount", "يجب أن يكون المبلغ غير صفر.");
                return View(cashTransaction);
            }

            // Validate cashbox balance for expenses
            if (cashTransaction.Type != "Payment" && cashTransaction.Amount < 0 && cashbox.Balance < -cashTransaction.Amount)
            {
                ModelState.AddModelError("Amount", "الخزنة لا تحتوي على رصيد كافٍ.");
                return View(cashTransaction);
            }

            cashbox.Balance += cashTransaction.Amount; // Positive for income, negative for expense
            _context.Update(cashbox);

            _context.CashboxTransactions.Add(cashTransaction);
            await _context.SaveChangesAsync();

            // Update related balances
            if (cashTransaction.Type == "Payment" && cashTransaction.CustomerId.HasValue)
            {
                var customer = await _context.Customers.FindAsync(cashTransaction.CustomerId.Value);
                customer.Balance -= cashTransaction.Amount;
                _context.Update(customer);
            }
            else if (cashTransaction.Type == "Payment" && cashTransaction.SupplierId.HasValue)
            {
                var supplier = await _context.Suppliers.FindAsync(cashTransaction.SupplierId.Value);
                supplier.Balance -= cashTransaction.Amount;
                _context.Update(supplier);
            }
            else if (cashTransaction.Type == "Salary" && cashTransaction.EmployeeId.HasValue)
            {
                // Optional: Update employee salary record if needed
            }

            await _context.SaveChangesAsync();
            return RedirectToAction("IndexCash");
        }

        ViewBag.Customers = _context.Customers.ToList();
        ViewBag.Suppliers = _context.Suppliers.ToList();
        ViewBag.Employees = _context.Employees.ToList();
        ViewBag.Cashboxes = _context.Cashboxes.ToList();
        return View(cashTransaction);
    }

    // Index for Transactions
    public async Task<IActionResult> Index()
    {
        var transactions = await _context.Transactions.ToListAsync();
        return View(transactions);
    }

    // Index for Cashbox Transactions
    public async Task<IActionResult> IndexCash()
    {
        var cashTransactions = await _context.CashboxTransactions.ToListAsync();
        return View(cashTransactions);
    }
}
